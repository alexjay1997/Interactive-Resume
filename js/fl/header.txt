        function showSlotMod(provider){
            const layer2 = document.getElementById('layer2');
            const modal_games = document.getElementById('modal_games');
            const slot_content = document.getElementById('slot_content');

            layer2.className = 'active';
            modal_games.className += ' active';

            slot_content.innerHTML = '';
            window.showLoader();
            fetch(`/api/v1/rubygames/list/bysubprovider?category=slotmachine&subProviderName=${provider}`, {
                method: 'GET',
            })
            .then(res => res.json())
            .then(data => {
                data.forEach(slot => {
                    const div = document.createElement('div');
                    div.setAttribute('data-favorite', 'off');
                    div.className = 'item';
                    div.innerHTML = `
                        <div class="img" style="height: 194px; border-radius: 10px !important;">
                            <img src="${slot.card_image}" alt="">
                            <a href="javascript:void(0)" onclick="pageLinkUrl('/game/${slot.id}')" >
                                <div class="figcaption game_link" data-link="/game/${slot.id}">
                                    <span>${slot.name_kr}</span>
                                </div>
                            </a>
                        </div> 
                        <div class="desc">
                            <span>${slot.name_kr}</span>
                        </div>
                    `;
                    slot_content.appendChild(div);
                });
                window.hideLoader();
            });
        }
